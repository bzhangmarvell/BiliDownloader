# 批量下载 UP 主视频 - 设计方案

**创建日期**: 2026-03-30  
**项目**: BiliDownloader  
**版本**: v1.0.6

---

## 一、需求概述

在现有单视频下载功能基础上，新增批量下载 UP 主所有视频的功能，支持：

1. ✅ 通过 UP 主 UID 或空间链接获取视频列表
2. ✅ 带 checkbox 的列表选择特定视频
3. ✅ 多种排序方式（投稿时间、视频名称、播放量、时长）
4. ✅ 过滤功能（关键词、日期范围、最大数量）
5. ✅ 批量下载进度管理
6. ✅ 断点续传支持

---

## 二、技术架构

### 2.1 核心 API

**获取 UP 主视频列表 API**:
```
GET https://api.bilibili.com/x/space/wbi/arc/search
参数:
  - mid: UP 主 UID
  - pn: 页码 (从 1 开始)
  - ps: 每页数量 (建议 30)
  - order: 排序方式 (pubdate/play/click)
```

**响应结构**:
```json
{
  "code": 0,
  "data": {
    "page": {
      "count": 523,      // 总视频数

      "vlist": [
        {
          "bvid": "BV1xx411c7mD",

          "title": "视频标题",

          "length": "05:23",
```

```typescript
// UP 主视频信息
export interface UpVideo {
  bvid: string;
  aid: number;
  title: string;
  cover: string;
  length: string;        // 时长字符串 "05:23"
  created: number;       // 发布时间戳
  play: number;          // 播放量
  cid?: number;
}

// UP 主信息
export interface UpInfo {
  mid: number;
  name: string;
  face: string;
  fans: number;
  videoCount: number;
}

// 排序方式
export type SortBy = 
  | 'created_desc'    // 最新优先
  | 'created_asc'     // 最旧优先
  | 'title_asc'       // 名称 A-Z
  | 'title_desc'      // 名称 Z-A
  | 'play_desc'       // 播放量高到低
  | 'play_asc'        // 播放量低到高
  | 'duration_desc'   // 时长长到短
  | 'duration_asc';   // 时长短到长

// 批量下载配置
export interface BatchDownloadOptions {
  mid: number;                    // UP 主 UID
  quality: number;                // 清晰度
  outputPath: string;             // 下载路径
  sortBy: SortBy;                 // 排序方式
  filterKeyword?: string;         // 关键词过滤
  dateRange?: {                   // 日期范围
    start?: number;               // 开始时间戳
    end?: number;                 // 结束时间戳
  };
  maxCount?: number;              // 最大下载数量 (0=全部)
  selectedVideos?: string[];      // 选中的视频 BV 号列表
}

// 批量任务
export interface BatchTask {
  id: string;
  mid: number;
  upName: string;
  upFace: string;
  totalVideos: number;            // 总视频数
  selectedCount: number;          // 选中的视频数
  downloadedCount: number;        // 已下载完成数
  failedCount: number;            // 失败数
  status: 'pending' | 'running' | 'paused' | 'completed' | 'error';
  progress: number;               // 整体进度 0-100
  currentTask?: string;           // 当前下载的任务 ID
  tasks: DownloadTask[];          // 子任务列表
  failedVideos: Array<{
    bvid: string;
    title: string;
    error: string;
  }>;
  createdAt: number;
  completedAt?: number;
}
```

### 3.2 API 实现 (`api.ts`)

```typescript
// 获取 UP 主信息（需要登录）
async getUpInfo(mid: number): Promise<UpInfo> {
  const data = await this.requestWithCookie<any>(
    'https://api.bilibili.com/x/space/acc/info',
    { mid },
    true // 需要 WBI 签名
  );
  
  return {
    mid: data.mid,
    name: data.name,
    face: data.face,
    fans: data.fans,
    videoCount: data.videos?.count || 0,
  };
}

// 获取 UP 主视频列表（单页，需要登录）
async getUpVideos(
  mid: number, 
  pn: number = 1, 
  ps: number = 30
): Promise<{
  videos: UpVideo[],
  total: number,
  hasMore: boolean
}> {
  const data = await this.requestWithCookie<any>(
    'https://api.bilibili.com/x/space/wbi/arc/search',
    { mid, pn, ps },
    true // 需要 WBI 签名
  );
  
  const videos: UpVideo[] = (data.list?.vlist || []).map((v: any) => ({
    bvid: v.bvid,
    aid: v.aid,
    title: v.title,
    cover: v.pic,
    length: v.length,
    created: v.created,
    play: v.play,
    cid: v.cid,
  }));
  
  const totalCount = data.page?.count || 0;
  
  return {
    videos,
    total: totalCount,
    hasMore: videos.length * pn < totalCount,
  };
}

// 获取 UP 主全部视频（自动分页）
async fetchAllUpVideos(mid: number, maxPage: number = 0): Promise<UpVideo[]> {
  const allVideos: UpVideo[] = [];
  let pn = 1;
  const ps = 30;
  
  while (true) {
    console.log(`[API] Fetching page ${pn} for UP ${mid}`);
    const { videos, total: _, hasMore } = await this.getUpVideos(mid, pn, ps);
    allVideos.push(...videos);
    
    // 检查是否继续
    if (!hasMore) break;
    if (maxPage > 0 && pn >= maxPage) break;
    
    pn++;
    // 避免请求过快，添加延迟
    await new Promise(resolve => setTimeout(resolve, 300));
  }
  
  console.log(`[API] Fetched ${allVideos.length} videos for UP ${mid}`);
  return allVideos;
}
```

### 3.3 下载引擎扩展 (`engine.ts`)

```typescript
// 批量任务管理
private batchTasks: Map<string, BatchTask> = new Map();

// 创建批量任务
async createBatchTask(options: BatchDownloadOptions): Promise<BatchTask> {
  // 1. 获取 UP 主信息
  const upInfo = await bilibiliAPI.getUpInfo(options.mid);
  
  // 2. 获取视频列表（自动分页）
  const maxPage = options.maxCount ? Math.ceil(options.maxCount / 30) : 0;
  const allVideos = await bilibiliAPI.fetchAllUpVideos(options.mid, maxPage);
  
  // 3. 应用过滤条件
  let filteredVideos = this.filterVideos(allVideos, options);
  
  // 4. 排序
  filteredVideos = this.sortVideos(filteredVideos, options.sortBy);
  
  // 5. 限制数量
  if (options.maxCount && options.maxCount > 0) {
    filteredVideos = filteredVideos.slice(0, options.maxCount);
  }
  
  // 6. 如果指定了选中的视频，只保留选中的
  if (options.selectedVideos && options.selectedVideos.length > 0) {
    const selectedSet = new Set(options.selectedVideos);
    filteredVideos = filteredVideos.filter(v => selectedSet.has(v.bvid));
  }
  
  // 7. 创建批量任务
  const batchTask: BatchTask = {
    id: `batch_${options.mid}_${Date.now()}`,
    mid: options.mid,
    upName: upInfo.name,
    upFace: upInfo.face,
    totalVideos: allVideos.length,
    selectedCount: filteredVideos.length,
    downloadedCount: 0,
    failedCount: 0,
    status: 'pending',
    progress: 0,
    tasks: [],
    failedVideos: [],
    createdAt: Date.now(),
  };
  
  // 8. 为每个视频创建子任务
  for (const video of filteredVideos) {
    const videoInfo = await bilibiliAPI.getVideoInfo(video.bvid);
    const task: DownloadTask = {
      id: `${video.bvid}_${videoInfo.cid}_${Date.now()}`,
      bvid: video.bvid,
      aid: video.aid,
      cid: videoInfo.cid,
      title: video.title,
      quality: options.quality,
      status: 'pending',
      progress: 0,
      speed: 0,
      outputPath: path.join(options.outputPath, upInfo.name, `${video.title} [${video.bvid}].mp4`),
      createdAt: Date.now(),
    };
    batchTask.tasks.push(task);
    this.tasks.set(task.id, task);
  }
  
  this.batchTasks.set(batchTask.id, batchTask);
  this.emit('batch-task-created', batchTask);
  
  return batchTask;
}

// 启动批量任务
async startBatchTask(batchId: string): Promise<void> {
  const batchTask = this.batchTasks.get(batchId);
  if (!batchTask) throw new Error(`Batch task ${batchId} not found`);
  
  batchTask.status = 'running';
  this.emit('batch-task-started', batchTask);
  
  // 顺序下载（避免 API 限流）
  for (const task of batchTask.tasks) {
    const currentBatchTask = this.batchTasks.get(batchId);
    if (!currentBatchTask || currentBatchTask.status === 'paused' || currentBatchTask.status === 'error') {
      break;
    }
    
    if (task.status === 'completed') {
      batchTask.downloadedCount++;
      continue;
    }
    
    batchTask.currentTask = task.id;
    this.emit('batch-task-progress', batchTask);
    
    try {
      await this.startTask(task.id);
      batchTask.downloadedCount++;
    } catch (error: any) {
      batchTask.failedCount++;
      batchTask.failedVideos.push({
        bvid: task.bvid,
        title: task.title,
        error: error.message,
      });
    }
    
    batchTask.progress = (batchTask.downloadedCount / batchTask.tasks.length) * 100;
    this.emit('batch-task-progress', batchTask);
    
    // 请求间隔，避免限流
    await new Promise(resolve => setTimeout(resolve, 500));
  }
  
  batchTask.status = 'completed';
  batchTask.completedAt = Date.now();
  batchTask.currentTask = undefined;
  this.emit('batch-task-completed', batchTask);
}

// 暂停批量任务
pauseBatchTask(batchId: string): void {
  const batchTask = this.batchTasks.get(batchId);
  if (batchTask && batchTask.status === 'running') {
    batchTask.status = 'paused';
    this.emit('batch-task-paused', batchTask);
  }
}

// 恢复批量任务
async resumeBatchTask(batchId: string): Promise<void> {
  const batchTask = this.batchTasks.get(batchId);
  if (batchTask && batchTask.status === 'paused') {
    batchTask.status = 'running';
    await this.startBatchTask(batchId);
  }
}

// 获取批量任务
getBatchTask(batchId: string): BatchTask | undefined {
  return this.batchTasks.get(batchId);
}

// 辅助方法：过滤视频
private filterVideos(videos: UpVideo[], options: BatchDownloadOptions): UpVideo[] {
  let filtered = videos;
  
  if (options.filterKeyword) {
    const keyword = options.filterKeyword.toLowerCase();
    filtered = filtered.filter(v => v.title.toLowerCase().includes(keyword));
  }
  
  if (options.dateRange) {
    const { start, end } = options.dateRange;
    filtered = filtered.filter(v => {
      if (start && v.created < start) return false;
      if (end && v.created > end) return false;
      return true;
    });
  }
  
  return filtered;
}

// 辅助方法：排序视频
private sortVideos(videos: UpVideo[], sortBy: string): UpVideo[] {
  return [...videos].sort((a, b) => {
    switch (sortBy) {
      case 'created_desc': return b.created - a.created;
      case 'created_asc': return a.created - b.created;
      case 'title_asc': return a.title.localeCompare(b.title, 'zh-CN');
      case 'title_desc': return b.title.localeCompare(a.title, 'zh-CN');
      case 'play_desc': return b.play - a.play;
      case 'play_asc': return a.play - b.play;
      case 'duration_desc': {
        const durA = this.parseDuration(a.length);
        const durB = this.parseDuration(b.length);
        return durB - durA;
      }
      case 'duration_asc': {
        const durA = this.parseDuration(a.length);
        const durB = this.parseDuration(b.length);
        return durA - durB;
      }
      default: return b.created - a.created;
    }
  });
}

// 辅助方法：解析时长字符串
private parseDuration(length: string): number {
  const parts = length.split(':').map(Number);
  if (parts.length === 2) return parts[0] * 60 + parts[1];
  else if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
  return 0;
}
```

## 六、技术难点与解决方案

### 6.1 API 限流问题

**问题**: B 站 API 有频率限制，快速获取大量视频可能触发限流

**解决方案**:
- 分页请求间隔 300ms
- 下载间隔 500ms
- 失败重试机制（3 次，指数退避）
- 使用 WBI 签名提高请求成功率

### 6.2 登录 Cookie

**问题**: B 站 API 需要登录 cookie 才能访问 UP 主视频列表

**解决方案**:
- 使用 `requestWithCookie` 方法
- 未登录时提示"请先登录 B 站账号"
- 扫码登录后自动保存 cookie

### 6.3 大列表性能

**问题**: UP 主视频可能上千个，渲染大量 DOM 影响性能

**解决方案**:
- 虚拟滚动（react-window）
- 分页加载（每页 50 个）
- 懒加载：先加载前 100 个，滚动到底部再加载更多
